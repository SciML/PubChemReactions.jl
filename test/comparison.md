u = symmatrix('u', [3 1])
x = symmatrix2sym(u)
eqs = [
x(1) == x(3),
x(1) + x(2) == 2 * x(3),
-x(1) + x(2) == 0,
x(1) > 0
]
solution = solve(eqs,[sym('u1'),sym('u2'),sym('u3')],...
'PrincipalValue',true,...
'ReturnConditions',true);

solution.u1, solution.u2, solution.u3 % (1,1,1)
